이프문의 사용법:
if(조건)//실행되는게 한줄실행일경우 {}없이 사용가능
{ }// 괄호안에 조건일 경우 실행할코드입력
else if(다른조건)// 윗조건 이외의 다른조건을 걸고싶을때사용
{}
else//윗조건 이외의 모든경우들을 처리할때 사용
{}

switch 문의 사용 

int main()
{
int a=0;
printf("0과 1 을 입력하시오 : ");
scanf("%d",&a);
switch(a)//a가 위에서 입력받은 것이여도 되지만 다른 연산결과를 원한다면 위에서 연산한결과를 a에다 넣으면 여러 응용이가능)
{
case 1:
printf("1을입력하셧네요");
break;//조건 충족시 위의 코드 실행후 내려가지않고 종료
case 0: 
printf("0을 입력하셧네요 ");
break;
}

}
스위치문사용의 경우 연산된값의 결과를 일일히 채킹 하면서 코드 실행이가능하고
이프문의경우 특정조건을 걸러내는 식으로 코드실행이가능하다.
