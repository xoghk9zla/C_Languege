#include <stdio.h>


int main()//이거 반복문비교
{
	//For문 사용법
	//for(초기화식;조건식;증감식)
	//	반복할 문장

	for(int i=0;i<10;i++)//장점 한줄로끝남ㅎ, 실행순서 초기화식->조건식->문장->증감식->조건식->문장->증감식->조건식->문장...
		printf("민규 ");
	
	printf("\n");

	//while문 사용법
	//while(조건식)
	//	반복할 문장
	int j=0;
	while(j<10){//조건식이 참이라면 문장을 계속 반복함, 실행순서 조건식->문장->조건식->문장...
		printf("민규 ");
		j++;
	}

	printf("\n");

	//do while
	//do while(조건식)
	//	반복할 문장
	
	int k=0;
	do{//먼저 실행한후 다음 실행부터는 조건식을 확인함, 실행순서 문장->조건식->문장....
		printf("민규 ");
		k++;
	}while(k<10);
}



// for문안에서 변수 선언은 c++에서 가능
